<script>
    function calcSolar() {
        const bill = document.getElementById('bill').value;
        if(bill > 0) {
            const kwRequired = (bill / 12) / 30 / 4.5; 
            const panels = Math.ceil((kwRequired * 1000) / 550);
            document.getElementById('sys-size').innerText = kwRequired.toFixed(1);
            document.getElementById('panels').innerText = panels;
        }
    }

    function calcCost() {
        const watts = document.getElementById('watts').value;
        const hours = document.getElementById('hours').value;
        if(watts > 0 && hours > 0) {
            const monthly = (watts / 1000) * hours * 12 * 30;
            document.getElementById('monthly-cost').innerText = '₱' + monthly.toLocaleString(undefined, {maximumFractionDigits: 0});
        }
    }

    function getGuide() {
        const type = document.getElementById('load-type').value;
        const res = document.getElementById('guide-res');
        const guides = {
            'res': '<b>Standard Outlet:</b> Use 3.5mm² THHN Wire + 20A Breaker.',
            'com': '<b>Heavy Load:</b> Use 5.5mm² or 8.0mm² + 30A-40A Breaker.',
            'sol': '<b>Solar Setup:</b> Requires dedicated DC breakers and 6mm² PV cables.',
            'light': '<b>Lighting:</b> Use 2.0mm² THHN Wire + 15A Breaker.'
        };
        res.innerHTML = guides[type] || 'Select to see PEC standards.';
    }
</script>
